-- AIMBOT + FOV CIRCLE (VÒNG TRÒN GIỚI HẠN) CHO DELTA EXECUTOR

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local Camera = game:GetService("Workspace").CurrentCamera
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local CoreGui = game:GetService("CoreGui")

-- Xóa giao diện cũ nếu có để tránh lỗi trùng lặp
if CoreGui:FindFirstChild("DeltaAimbotFOVUI") then
    CoreGui.DeltaAimbotFOVUI:Destroy()
end

-- --- CẤU HÌNH TRỰC QUAN ---
local AimbotEnabled = false  -- Mặc định tắt
local TeamCheck = true       -- Bật true để không bắn đồng đội (false để khóa tất cả)
local FOV_Radius = 150       -- Kích thước vòng tròn (bạn có thể chỉnh to nhỏ tùy ý)

-- --- TẠO VÒNG TRÒN FOV GIỮA MÀN HÌNH BẰNG GIAO DIỆN PHẲNG ---
-- Sử dụng cách này để đảm bảo chạy mượt trên cả Android/Delta không bị lỗi Drawing
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "DeltaAimbotFOVUI"
ScreenGui.Parent = CoreGui
ScreenGui.ResetOnSpawn = false

local FOVFrame = Instance.new("Frame")
FOVFrame.Name = "FOV_Circle"
FOVFrame.AnchorPoint = Vector2.new(0.5, 0.5)
FOVFrame.Position = UDim2.new(0.5, 0, 0.5, 0) -- Giữa màn hình
FOVFrame.Size = UDim2.new(0, FOV_Radius * 2, 0, FOV_Radius * 2)
FOVFrame.BackgroundTransparency = 1 -- Làm mờ nền trong suốt
FOVFrame.Visible = false -- Chỉ hiện khi bật aimbot
FOVFrame.Parent = ScreenGui

-- Tạo viền tròn cho FOV
local FOVCorner = Instance.new("UICorner")
FOVCorner.CornerRadius = UDim.new(1, 0) -- Biến hình vuông thành hình tròn tuyệt đối
FOVCorner.Parent = FOVFrame

local FOVStroke = Instance.new("UIStroke")
FOVStroke.Color = Color3.fromRGB(0, 255, 255) -- Vòng tròn màu xanh neon rực rỡ
FOVStroke.Thickness = 1.5
FOVStroke.Transparency = 0.4
FOVStroke.Parent = FOVFrame

-- --- TẠO NÚT BẬT/TẮT TRÊN MÀN HÌNH (DRAGGABLE) ---
local ToggleButton = Instance.new("TextButton")
ToggleButton.Name = "AimbotToggle"
ToggleButton.Size = UDim2.new(0, 120, 0, 45)
ToggleButton.Position = UDim2.new(0.1, 0, 0.1, 0)
ToggleButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50) -- Đỏ khi tắt
ToggleButton.Text = "Aimbot FOV: OFF"
ToggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
ToggleButton.Font = Enum.Font.SourceSansBold
ToggleButton.TextSize = 14
ToggleButton.Active = true
ToggleButton.Draggable = true -- Giữ ngón tay để kéo di chuyển nút trên điện thoại
ToggleButton.Parent = ScreenGui

local ToggleCorner = Instance.new("UICorner")
ToggleCorner.CornerRadius = UDim.new(0, 8)
ToggleCorner.Parent = ToggleButton

local ToggleStroke = Instance.new("UIStroke")
ToggleStroke.Color = Color3.fromRGB(255, 255, 255)
ToggleStroke.Thickness = 1.5
ToggleStroke.Parent = ToggleButton

-- Sự kiện Click Bật/Tắt
ToggleButton.MouseButton1Click:Connect(function()
    AimbotEnabled = not AimbotEnabled
    FOVFrame.Visible = AimbotEnabled -- Hiện vòng tròn khi bật, ẩn khi tắt
    if AimbotEnabled then
        ToggleButton.Text = "Aimbot FOV: ON"
        ToggleButton.BackgroundColor3 = Color3.fromRGB(50, 180, 80) -- Xanh khi bật
    else
        ToggleButton.Text = "Aimbot FOV: OFF"
        ToggleButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
    end
end)

-- --- HÀM TÌM KẺ ĐỊCH GẦN TÂM NHẤT NẰM TRONG VÒNG TRÒN FOV ---
local function GetClosestPlayerInFOV()
    local ClosestPlayer = nil
    local ShortestDistance = FOV_Radius -- Khoảng cách tối đa chính là bán kính vòng tròn
    local CenterScreen = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)

    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character then
            -- Kiểm tra nếu bật TeamCheck
            if not TeamCheck or (Player.Team ~= LocalPlayer.Team) then
                local Head = Player.Character:FindFirstChild("Head")
                local Humanoid = Player.Character:FindFirstChildOfClass("Humanoid")
                
                if Head and Humanoid and Humanoid.Health > 0 then
                    -- Chuyển tọa độ 3D của Đầu kẻ địch thành tọa độ 2D trên màn hình
                    local ScreenPos, OnScreen = Camera:WorldToViewportPoint(Head.Position)
                    
                    if OnScreen then
                        local EnemyXY = Vector2.new(ScreenPos.X, ScreenPos.Y)
                        -- Tính khoảng cách từ Đầu kẻ địch tới chính giữa tâm vòng tròn
                        local Distance = (EnemyXY - CenterScreen).Magnitude
                        
                        -- Nếu khoảng cách nhỏ hơn bán kính vòng tròn (tức là nằm bên trong FOV)
                        if Distance < ShortestDistance then
                            ClosestPlayer = Player
                            ShortestDistance = Distance -- Cập nhật để tìm kẻ gần tâm nhất
                        end
                    end
                end
            end
        end
    end
    return ClosestPlayer
end

-- --- VÒNG LẶP KHÓA TÂM LIÊN TỤC (KHÔNG LAG) ---
RunService.RenderStepped:Connect(function()
    -- Cập nhật lại kích thước FOV liên tục đề phòng bạn đổi kích thước màn hình ngang/dọc
    FOVFrame.Size = UDim2.new(0, FOV_Radius * 2, 0, FOV_Radius * 2)

    if AimbotEnabled then
        local Target = GetClosestPlayerInFOV()
        if Target and Target.Character and Target.Character:FindFirstChild("Head") then
            -- Khóa thẳng góc nhìn Camera vào Đầu kẻ địch
            Camera.CFrame = CFrame.new(Camera.CFrame.Position, Target.Character.Head.Position)
        end
    end
end)

print("Aimbot FOV cho Delta đã sẵn sàng!")
